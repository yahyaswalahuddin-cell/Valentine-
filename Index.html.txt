  <script>
    const noBtn = document.getElementById("noBtn");
    const yesBtn = document.getElementById("yesBtn");
    const questionCard = document.getElementById("questionCard");
    const celebrationCard = document.getElementById("celebrationCard");

    function moveNoButton() {
      const card = questionCard.getBoundingClientRect();
      const maxX = card.width - noBtn.offsetWidth;
      const maxY = card.height - noBtn.offsetHeight;

      noBtn.style.left = Math.random() * maxX + "px";
      noBtn.style.top = Math.random() * maxY + "px";
    }

    noBtn.addEventListener("mouseover", moveNoButton);
    noBtn.addEventListener("click", moveNoButton);

    yesBtn.addEventListener("click", () => {
      questionCard.classList.add("hidden");
      celebrationCard.classList.remove("hidden");
    });

    moveNoButton();
